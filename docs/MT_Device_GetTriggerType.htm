Datalogic .NET SDK

GetTriggerType (Method)

 

Syntax

 

   public static TriggerInputType GetTriggerType(TriggerId trigger);

 

 

Class                           : Device

Namespace                : Datalogic.API

Assembly                   : Datalogic.API.DLL

 

 

Description

Returns the type of input data acquired when a trigger is activated.

 

 

Parameters Table

 

Type

Parameter

Description

[in]

TriggerId

trigger

Specifies the trigger to query.

 

 

 

 

[returns]

TriggerInputType

 

The return value is the input data type currently set for acquisition when the indicated trigger is activated.

[on error]

Device.TriggerInputType.None

 

If the trigger is disabled.

 

 

Example

 

        private void btnTriggerType_Click(object sender, EventArgs e)

        {

            bool b = false;

 

            Device.TriggerId ENUM_TriggId;

            Device.TriggerInputType ENUM_GetTriggInputType;

 

            switch(comboBox1.SelectedIndex)

            {

                case 0: ENUM_TriggId = Device.TriggerId.LeftSide; break;

                case 1: ENUM_TriggId = Device.TriggerId.Pistol; break;

                case 2: ENUM_TriggId = Device.TriggerId.RightSide; break;

                case 3: ENUM_TriggId = Device.TriggerId.Scan; break;

                default:

                    ENUM_TriggId = Device.TriggerId.Scan;

                    break;

            }

            b = Device.SetTriggerType(ENUM_TriggId, Device.TriggerInputType.Barcode);

 

            ENUM_GetTriggInputType = Device.GetTriggerType(ENUM_TriggId);

           

            txtGetTrInType.Text = ENUM_GetTriggInputType.ToString();

 

        }
