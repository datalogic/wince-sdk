WiFiAddConfig

Adds a new Wi-Fi connection profile or modifies one that already exists.

 

SDCERR WiFiAddConfig(SDCConfig *cfg);

 

Parameters

SDCConfig *cfg

Settings for the desired profile

Return Value

SDCERR_SUCCESS on success.

SDCERR_INVALID_CONFIG if configuration is bad

 

 

 

Example: Adding a WPA2-PSK network

 

  SDCConfig cfg;

  memset(&cfg, 0, sizeof(SDCConfig));

 

  strcpy(cfg.SSID, "myNetwork");

  strcpy(cfg.configName, "myNetwork");

  strcpy(cfg.PSK.buffer, "myPassword");

  cfg.PSK.size = strlen(cfg.PSK.buffer)+1;

  cfg.authType = WPA2PSK;

  cfg.encr = AES;

  cfg.Broadcast = TRUE;

 

  SDCERR err = WiFiAddConfig(&cfg);

  printf("result=%d\n", err);

Example: Adding a PEAP Network

 

SDCConfig cfg = {}; // zero out. alternatively: memset(&cfg, 0, sizeof(cfg));

PEAPConfig peapCfg = {};

cfg.PEAP = &peapCfg;

 

strcpy(cfg.configName, "peap");

strcpy(cfg.SSID, SSID_EAPPEAP);

cfg.Broadcast = TRUE;

cfg.EapType = PPP_EAP_PEAP;

cfg.PEAP->Cred.CredType = PASSWORD;

cfg.PEAP->Cred.CredSource = FROM_PROFILE;

strcpy(cfg.PEAP->Cred.UserName, "eappeap");

strcpy(cfg.PEAP->Cred.Password, "eappeap");

cfg.PEAP->InnerMethod = EAP_MsChapV2;

 

SDCERR err = WiFiAddConfig(&cfg);

printf("result=%d\n", err);
