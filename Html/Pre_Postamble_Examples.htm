Preamble/Postamble Examples

The following procedures give an example for:

  1. setting a postamble;

  2. getting a postamble;

  3. setting a preamble;

  4. getting a preamble.

 

 

Setting a Postamble

This example allows appending the carriage return character ([CR]) to the trailing barcode character.

 

// Windows Main Header File

#include <windows.h>

 

//Text Function Main Header File

#include <tchar.h>

 

//Serial Bar Code Raster Reader PARAMETERS Enumeration

#include “BBCR_Parameters.h”

 

//One Dimensional Bar Code Reader PARAMETERS Enumeration

#include “BCD_Parameters.h”

 

//DATALOGIC SCANNER SETUP CLASS Header File

#include “DL_SEDAPI.h”

 

bool SetPostamble (LPCTSRT szText)

{

       bool bResponse = false;

 

       DLScannerSetup oSetup;

 

       #pragma pack (1)

       struct PostambleString

       {

       unsigned long m_nLength;

       TCHAR m_sText[_MAX_PATH];

       };

 

       PostambleString oText;

 

       unsigned long nLength = (_tcslen(szText)<=_MAX_PATH)? tcslen(szText):MAX_PATH;

 

       memcpy(oText.m_sText, szText, nLength*sizeof(TCHAR));

 

       oText.m_nLength = nLength;

 

       int nParamaID = -1;

 

       if (oSetup.getReaderIdentifier() == SE_READER_BBCR_CLASS_ID)

       {

              nParamID = BBCR_PARAM_POSTAMBLE_STRING;

       }

else if (oSetup.getReaderIdentifier() == SE_READER_BCD_CLASS_ID)

       {

              nParamID = BCD_PARAM_POSTAMBLE_STRING;

       }

       if (nParamID != -1)

       {

              oSetup.setParameterByReference (nParamID, &oText);

 

              if (oSetup.getStatus() == DLScannerSetup::s_nOK)

              {

                    bResponse = true;

              }

       }

       return (bResponse);

}

 

int WINAPI WinMain(              HINSTANCE hInstance,

                    HINSTANCE hPrevInstance,

                    LPTSTR    lpCmdLine,

                    int       nCmdShow)

{

       SetPostamble(_T(“\r\n”));

 

       return 0;

}

 

 

Getting a Postamble

This example allows showing the postamble text in a message box.

 

// Windows Main Header File

#include <windows.h>

 

//Text Function Main Header File

#include <tchar.h>

 

//Serial Bar Code Raster Reader PARAMETERS Enumeration

#include “BBCR_Parameters.h”

 

//One Dimensional Bar Code Reader PARAMETERS Enumeration

#include “BCD_Parameters.h”

 

//DATALOGIC SCANNER SETUP CLASS Header File

#include “DL_SEDAPI.h”

 

bool GetPostamble (LPTSTR szDest, unsigned long nDestCapacity)

{

       bool bResponse = false;

 

       DLScannerSetup oSetup;

 

       #pragma pack (1)

       struct PostambleString

       {

       unsigned long m_nLength;

       TCHAR m_sText[_MAX_PATH];

       };

 

       PostambleString oText;

 

       oText.m_nLength =_MAX_PATH;

 

       int nParamaID = -1;

 

       if (oSetup.getReaderIdentifier() == SE_READER_BBCR_CLASS_ID)

       {

              nParamID = BBCR_PARAM_POSTAMBLE_STRING;

       }

else if (oSetup.getReaderIdentifier() == SE_READER_BCD_CLASS_ID)

       {

              nParamID = BCD_PARAM_POSTAMBLE_STRING;

       }

       if (nParamID != -1)

       {

              oSetup.getParameterByReference (nParamID, &oText);

 

              if (oSetup.getStatus() == DLScannerSetup::s_nOK)

              {

                    if (nDestCapacity>oText.m_nLength)

                    {

bResponse = true;

memcpy(szDest, oText.m_sText, oText.m_nLength*sizeof(TCHAR));

szDest[oText.m_nLength] = _T(‘\0’);

}

              }

}

       return (bResponse);

}

 

int WINAPI WinMain(              HINSTANCE hInstance,

                    HINSTANCE hPrevInstance,

                    LPTSTR    lpCmdLine,

                    int       nCmdShow)

{

       TCHAR szMyString[MAX_PATH];

       if (GetPostamble(szMyString, MAX_PATH)==true)

       {

              MessageBox(NULL,szMyString, _T(“Postamble”), MB_OK);

}

 

return 0;

}
