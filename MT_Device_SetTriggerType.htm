SetTriggerType (Method)

 

Syntax

 

   public static Boolean SetTriggerType(TriggerId trigger, TriggerInputType inputType);

 

 

Class                           : Device

Namespace                : Datalogic.API

Assembly                   : Datalogic.API.DLL

 

 

Description

Sets the type of input data acquired when a trigger is activated.

 

 

Parameters Table

 

Type

Parameter

Description

[in]

TriggerId

trigger

Specifies the trigger to set.

[in]

TriggerInputType

inputType

Specifies the input data type to begin acquiring when the trigger is activated.

Device.TriggerInputType.None may also be used to disable the trigger.

 

 

 

 

[returns]

Boolean

 

TRUE is returned on success, otherwise FALSE is returned.

[on error]

FALSE

  

Returns false if it is specified an invalid trigger or input type.

 

 

Example

 

        private void btnTriggerType_Click(object sender, EventArgs e)

        {

            bool b = false;

 

            Device.TriggerId ENUM_TriggId;

            Device.TriggerInputType ENUM_GetTriggInputType;

 

            switch(comboBox1.SelectedIndex)

            {

                case 0: ENUM_TriggId = Device.TriggerId.LeftSide; break;

                case 1: ENUM_TriggId = Device.TriggerId.Pistol; break;

                case 2: ENUM_TriggId = Device.TriggerId.RightSide; break;

                case 3: ENUM_TriggId = Device.TriggerId.Scan; break;

                default:

                    ENUM_TriggId = Device.TriggerId.Scan;

                    break;

            }

            b = Device.SetTriggerType(ENUM_TriggId, Device.TriggerInputType.Barcode);

 

            ENUM_GetTriggInputType = Device.GetTriggerType(ENUM_TriggId);

           

            txtGetTrInType.Text = ENUM_GetTriggInputType.ToString();

 

        }
